# Color Roles

Semantic 토큰은 색상의 사용 목적에 따라 정의됩니다. Palette 토큰을 참조하여 실제 UI에서 사용됩니다.

---

## Background

Background 색상은 페이지, 카드, 컴포넌트의 배경에 사용됩니다.

### Tokens

import palette from '../../../../tokens/colors.js';

export const ColorChip = ({ color, isAlpha }) => {
  const baseStyle = {
    width: '24px',
    height: '24px',
    borderRadius: '4px',
    border: '1px solid #E0E0E0',
    display: 'inline-block',
    verticalAlign: 'middle',
    flexShrink: 0,
  };

  if (isAlpha) {
    return (
      <span style={{
        ...baseStyle,
        backgroundImage: `linear-gradient(45deg, #ccc 25%, transparent 25%),
                          linear-gradient(-45deg, #ccc 25%, transparent 25%),
                          linear-gradient(45deg, transparent 75%, #ccc 75%),
                          linear-gradient(-45deg, transparent 75%, #ccc 75%)`,
        backgroundSize: '8px 8px',
        backgroundPosition: '0 0, 0 4px, 4px -4px, -4px 0px',
        position: 'relative',
      }}>
        <span style={{
          position: 'absolute',
          inset: 0,
          borderRadius: '3px',
          background: color,
        }} />
      </span>
    );
  }

  return <span style={{ ...baseStyle, background: color }} />;
};

export const TokenLink = ({ href, children }) => (
  <a href={href} style={{
    color: '#111111',
    textDecoration: 'underline',
    fontFamily: "'SF Mono', Monaco, 'Courier New', monospace",
    fontSize: '14px',
  }}>
    {children}
  </a>
);

  </span>
);
